--GO TO MIRROR SERVER
--CREATE KEY
USE master;
CREATE MASTER KEY ENCRYPTION BY PASSWORD = 'TESTCERT7025';
GO

--CREATE CERT
USE master;
CREATE CERTIFICATE BS_MIRROR_cert 
   WITH SUBJECT = 'BS_MIRROR_cert';
GO

--CREATE ENDPOINT
CREATE ENDPOINT BS_MIRROR_ENDPOINT
   STATE = STARTED
   AS TCP (
      LISTENER_PORT=7025
      , LISTENER_IP = ALL
   ) 
   FOR DATABASE_MIRRORING ( 
      AUTHENTICATION = CERTIFICATE BS_MIRROR_cert
      , ENCRYPTION = REQUIRED ALGORITHM AES
      , ROLE = ALL
   );
GO

--BACKUP CERTIFICATE
BACKUP CERTIFICATE BS_MIRROR_cert TO FILE = 'C:\BS_MIRROR_cert.cer';
GO
