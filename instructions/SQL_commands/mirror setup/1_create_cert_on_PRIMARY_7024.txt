--GO TO PRIMARY SERVER
--CREATE KEY
USE master;
CREATE MASTER KEY ENCRYPTION BY PASSWORD = 'TESTCERT7024';
GO

--CREATE CERT
USE master;
CREATE CERTIFICATE BS_PRIMARY_cert 
   WITH SUBJECT = 'BS_PRIMARY_cert';
GO

--CREATE ENDPOINT
CREATE ENDPOINT BS_PRIMARY_ENDPOINT
   STATE = STARTED
   AS TCP (
      LISTENER_PORT=7024
      , LISTENER_IP = ALL
   ) 
   FOR DATABASE_MIRRORING ( 
      AUTHENTICATION = CERTIFICATE BS_PRIMARY_cert
      , ENCRYPTION = REQUIRED ALGORITHM AES
      , ROLE = ALL
   );
GO

--BACKUP CERTIFICATE
BACKUP CERTIFICATE BS_PRIMARY_cert TO FILE = 'C:\BS_PRIMARY_cert.cer';
GO
